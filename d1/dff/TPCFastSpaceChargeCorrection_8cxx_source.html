120 {

123  FlatObject::setActualBufferAddress(actualFlatBufferPtr);
124 
125  size_t rowsOffset = 0;
126  size_t rowsSize = sizeof(RowSplineInfo) * mGeo.getNumberOfRows();
127 
128  mRowSplineInfoPtr = reinterpret_cast<RowSplineInfo*>(mFlatBufferPtr + rowsOffset);
129 
130  size_t scOffset = alignSize(rowsOffset + rowsSize, SplineType::getClassAlignmentBytes());
131  size_t scSize = sizeof(SplineType) * mNumberOfScenarios;
132 
133  mScenarioPtr = reinterpret_cast<SplineType*>(mFlatBufferPtr + scOffset);

135  size_t scBufferOffset = alignSize(scOffset + scSize, SplineType::getBufferAlignmentBytes());
136  size_t scBufferSize = 0;
137 
138  for (int i = 0; i < mNumberOfScenarios; i++) {
139  SplineType& sp = mScenarioPtr[i];
140  sp.setActualBufferAddress(mFlatBufferPtr + scBufferOffset + scBufferSize);
141  scBufferSize = alignSize(scBufferSize + sp.getFlatBufferSize(), sp.getBufferAlignmentBytes());
142  }
143  size_t dataAlignment = SplineType::getParameterAlignmentBytes();
144  size_t sliceDataOffset = alignSize(scBufferOffset + scBufferSize, dataAlignment);
145 
146  mSplineData = reinterpret_cast<char*>(mFlatBufferPtr + sliceDataOffset);

232 void TPCFastSpaceChargeCorrection::finishConstruction()
233 {
235 
236  assert(mConstructionMask & ConstructionState::InProgress);
237 
238  for (int i = 0; i < mGeo.getNumberOfRows(); i++) {
239  assert(mConstructionRowSplineInfos[i].splineScenarioID >= 0);
240  }
241  for (int i = 0; i < mNumberOfScenarios; i++) {
242  assert(mConstructionScenarios[i].isConstructed());
243  }
244 
245  // organize memory for the flat buffer and caculate its size
246 
247  size_t rowsOffset = 0;
248  size_t rowsSize = sizeof(RowSplineInfo) * mGeo.getNumberOfRows();
249 
250  size_t scOffset = alignSize(rowsOffset + rowsSize, SplineType::getClassAlignmentBytes());
251  size_t scSize = sizeof(SplineType) * mNumberOfScenarios;
252 
253  size_t scBufferOffsets[mNumberOfScenarios];
254 
255  scBufferOffsets[0] = alignSize(scOffset + scSize, SplineType::getBufferAlignmentBytes());
256  size_t scBufferSize = 0;
257  for (int i = 0; i < mNumberOfScenarios; i++) {
258  SplineType& sp = mConstructionScenarios[i];
259  scBufferOffsets[i] = scBufferOffsets[0] + scBufferSize;
260  scBufferSize = alignSize(scBufferSize + sp.getFlatBufferSize(), sp.getBufferAlignmentBytes());
261  }
262 
263  size_t sliceDataOffset = alignSize(scBufferOffsets[0] + scBufferSize, SplineType::getParameterAlignmentBytes());
264 
265  mSliceDataSizeBytes = 0;
266  for (int i = 0; i < mGeo.getNumberOfRows(); i++) {
267  RowSplineInfo& row = mConstructionRowSplineInfos[i];
268  row.dataOffsetBytes = mSliceDataSizeBytes;
269  SplineType& sp = mConstructionScenarios[row.splineScenarioID];
270  mSliceDataSizeBytes += sp.getSizeOfParameters();
271  mSliceDataSizeBytes = alignSize(mSliceDataSizeBytes, SplineType::getParameterAlignmentBytes());
272  }
273 
274  FlatObject::finishConstruction(sliceDataOffset + mSliceDataSizeBytes * mGeo.getNumberOfSlices());
275 
276  mRowSplineInfoPtr = reinterpret_cast<RowSplineInfo*>(mFlatBufferPtr + rowsOffset);
277  for (int i = 0; i < mGeo.getNumberOfRows(); i++) {
278  mRowSplineInfoPtr[i] = mConstructionRowSplineInfos[i];
279  }
280 
281  mScenarioPtr = reinterpret_cast<SplineType*>(mFlatBufferPtr + scOffset);
282 
283  for (int i = 0; i < mNumberOfScenarios; i++) {
284  SplineType& sp0 = mConstructionScenarios[i];
285  SplineType& sp1 = mScenarioPtr[i];
286  new (&sp1) SplineType(); // first, call a constructor
287  sp1.cloneFromObject(sp0, mFlatBufferPtr + scBufferOffsets[i]);
288  }
289 
290  mSplineData = reinterpret_cast<char*>(mFlatBufferPtr + sliceDataOffset);
291 
292  releaseConstructionMemory();
293 
294  mTimeStamp = -1;
295 
296  // initialise all corrections to 0.
297 
298  for (int slice = 0; slice < mGeo.getNumberOfSlices(); slice++) {
299  for (int row = 0; row < mGeo.getNumberOfRows(); row++) {
300  const SplineType& spline = getSpline(slice, row);
301  float* data = getSplineData(slice, row);
302  for (int i = 0; i < spline.getNumberOfParameters(); i++) {
303  data[i] = 0.f;
304  }
305  }
306  }
307 }
