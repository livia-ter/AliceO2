158 uint64_t eightBytes(gsl::span<const std::byte> buffer)
159 {
160  return (static_cast<uint64_t>(buffer[0])) |
161  (static_cast<uint64_t>(buffer[1]) << 8) |
162  (static_cast<uint64_t>(buffer[2]) << 16) |
163  (static_cast<uint64_t>(buffer[3]) << 24) |
164  (static_cast<uint64_t>(buffer[4]) << 32) |
165  (static_cast<uint64_t>(buffer[5]) << 40) |

170 template <>
171 RAWDataHeaderV4 createRDH(gsl::span<const std::byte> buffer)
172 {
173  if (buffer.size() < 64) {
174  throw std::invalid_argument("buffer should be at least 64 bytes");
175  }
176  RAWDataHeaderV4 rdh;
177  rdh.word0 = eightBytes(buffer.subspan(0));
178  rdh.word1 = eightBytes(buffer.subspan(8));
179  rdh.word2 = eightBytes(buffer.subspan(16));
180  rdh.word3 = eightBytes(buffer.subspan(24));
181  rdh.word4 = eightBytes(buffer.subspan(32));
182  rdh.word5 = eightBytes(buffer.subspan(40));
183  rdh.word6 = eightBytes(buffer.subspan(48));
184  rdh.word7 = eightBytes(buffer.subspan(56));
185  return rdh;
186 }

188 template <>
189 RAWDataHeaderV4 createRDH(uint16_t cruId, uint8_t endpoint, uint8_t linkId, uint16_t feeId, uint32_t orbit, uint16_t bunchCrossing,
190  uint16_t payloadSize)
191 {
192  RAWDataHeaderV4 rdh;
193 
194  if (payloadSize > 0x0FFFF - sizeof(rdh)) {
195  throw std::invalid_argument("payload too big");
196  }
197 
198  uint16_t memorySize = payloadSize + sizeof(rdh);
199 
200  rdh.cruID = cruId;
201  rdhLinkId(rdh, linkId);
202  rdhEndpoint(rdh, endpoint);
203  rdh.feeId = feeId;
204  rdh.priority = 0;
205  rdh.blockLength = memorySize - sizeof(rdh); // FIXME: the blockLength disappears in RDHv5 ?
206  rdh.memorySize = memorySize;
207  rdh.offsetToNext = memorySize;
208  rdh.packetCounter = 0;
209  rdh.triggerType = 0;
210  rdh.detectorField = 0; // FIXME: fill this ?
211  rdh.par = 0; // FIXME: fill this ?
212  rdh.stop = 0;
213  rdh.pageCnt = 0;
214  rdh.heartbeatOrbit = orbit;
215  rdh.heartbeatBC = bunchCrossing;
216 
217  return rdh;
218 }

235 template <typename RDH>
236 int forEachRDH(gsl::span<uint8_t> buffer, std::function<void(RDH&, gsl::span<uint8_t>::size_type offset)> f)
237 {
238  int index{0};
239  int nrdh{0};
240  while (index < buffer.size()) {
241  RDH* rdhPtr = reinterpret_cast<RDH*>(&buffer[index]);
242  RDH& rdh = *rdhPtr;
243  if (!isValid(rdh)) {
244  break;
245  }
246  nrdh++;
247  if (f) {
248  f(rdh, index);
249  }
250  if (rdh.offsetToNext == 0) {
251  return -1;
252  }
253  index += rdh.offsetToNext;
254  }
255  return nrdh;
256 }
