50 void DecoderLogic::setClusterSize(uint16_t value)
51 {
52  mClusterSize = value;
53  int checkSize = mClusterSize + 2 - mSampaHeader.nof10BitWords();
54  mErrorMessage = std::nullopt;
55  if (mClusterSize == 0) {
56  mErrorMessage = "cluster size is zero";
57  }
58  if (checkSize > 0) {
59  mErrorMessage = "cluster size bigger than nof10BitWords";
60  }
61 #ifdef ULDEBUG
62  debugHeader() << " -> size=" << mClusterSize << " maskIndex=" << mMaskIndex
63  << " nof10BitWords=" << mSampaHeader.nof10BitWords()
64  << " " << (hasError() ? "ERROR:" : "") << errorMessage() << "\n";
65 #endif

107 uint16_t DecoderLogic::pop10()

142 template <>
143 void DecoderLogic::addSample<ChargeSumMode>(uint16_t sample)
144 {
145 #ifdef ULDEBUG
146  debugHeader() << "charge sum part = " << sample << "\n";
147 #endif
148  mSamples.emplace_back(sample);
149 
150  if (mClusterSize == 0) {
151  if (mSamples.size() != 2) {
152  throw std::invalid_argument(fmt::format("expected sample size to be 2 but it is {}", mSamples.size()));
153  }
154  uint32_t q = (((static_cast<uint32_t>(mSamples[1]) & 0x3FF) << 10) | (static_cast<uint32_t>(mSamples[0]) & 0x3FF));
155  SampaCluster sc(mClusterTime, q);
156  sendCluster(sc);
157  mSamples.clear();
158  }
159 }

161 void DecoderLogic::completeHeader()
162 {
163  uint64_t header{0};
164  for (auto i = 0; i < mHeaderParts.size(); i++) {
165  header += (static_cast<uint64_t>(mHeaderParts[i]) << (10 * i));
166  }
167 
168  mSampaHeader = SampaHeader(header);
169  mNof10BitWords = mSampaHeader.nof10BitWords();
170 
171 #ifdef ULDEBUG
172  debugHeader()
173  << fmt::format(">>>>> completeHeader {:013X}\n", header)
174  << "\n"
175  << mSampaHeader << "\n";
176 #endif
177 
178  mHeaderParts.clear();
179 }
