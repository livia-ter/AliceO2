114 template <typename T, typename ChunkingPolicy = Chunked>
115 class ColumnIterator : ChunkingPolicy
116 {
117  static constexpr char SCALE_FACTOR = std::is_same_v<std::decay_t<T>, bool> ? 3 : 0;
118 
119  public:
124  ColumnIterator(BackendColumnType const* column)
125  : mColumn{column},
126  mCurrentPos{nullptr},
127  mFirstIndex{0},
128  mCurrentChunk{0}
129  {
130  auto chunks = framework::getBackendColumnPtrData<BackendColumnType>(mColumn);
131  auto array = std::static_pointer_cast<arrow_array_for_t<T>>(chunks->chunk(mCurrentChunk));
132  mCurrent = reinterpret_cast<T const*>(array->values()->data()) + array->offset();
133  mLast = mCurrent + array->length();
134  }
135 
136  ColumnIterator() = default;
137  ColumnIterator(ColumnIterator<T, ChunkingPolicy> const&) = default;
138  ColumnIterator<T, ChunkingPolicy>& operator=(ColumnIterator<T, ChunkingPolicy> const&) = default;
139 
140  ColumnIterator(ColumnIterator<T, ChunkingPolicy>&&) = default;
141  ColumnIterator<T, ChunkingPolicy>& operator=(ColumnIterator<T, ChunkingPolicy>&&) = default;
142 
144  void nextChunk() const
145  {
146  auto chunks = framework::getBackendColumnPtrData<BackendColumnType>(mColumn);
147  auto previousArray = std::static_pointer_cast<arrow_array_for_t<T>>(chunks->chunk(mCurrentChunk));
148  mFirstIndex += previousArray->length();
149  mCurrentChunk++;
150  auto array = std::static_pointer_cast<arrow_array_for_t<T>>(chunks->chunk(mCurrentChunk));
151  mCurrent = reinterpret_cast<T const*>(array->values()->data()) + array->offset() - (mFirstIndex >> SCALE_FACTOR);
152  mLast = mCurrent + array->length() + (mFirstIndex >> SCALE_FACTOR);
153  }
154 
155  void prevChunk() const
156  {
157  auto chunks = framework::getBackendColumnPtrData<BackendColumnType>(mColumn);
158  auto previousArray = std::static_pointer_cast<arrow_array_for_t<T>>(chunks->chunk(mCurrentChunk));
159  mFirstIndex -= previousArray->length();
160  mCurrentChunk--;
161  auto array = std::static_pointer_cast<arrow_array_for_t<T>>(chunks->chunk(mCurrentChunk));
162  mCurrent = reinterpret_cast<T const*>(array->values()->data()) + array->offset() - (mFirstIndex >> SCALE_FACTOR);
163  mLast = mCurrent + array->length() + (mFirstIndex >> SCALE_FACTOR);
164  }
165 
166  void moveToChunk(int chunk)
167  {
168  if (mCurrentChunk < chunk) {
169  while (mCurrentChunk != chunk) {
170  nextChunk();
171  }
172  } else {
173  while (mCurrentChunk != chunk) {
174  prevChunk();
175  }
176  }
177  }
178 
180  void moveToEnd()
181  {
182  auto chunks = framework::getBackendColumnPtrData<BackendColumnType>(mColumn);
183  mCurrentChunk = chunks->num_chunks() - 1;
184  auto array = std::static_pointer_cast<arrow_array_for_t<T>>(chunks->chunk(mCurrentChunk));
185  assert(array.get());
186  mFirstIndex = mColumn->length() - array->length();
187  mCurrent = reinterpret_cast<T const*>(array->values()->data()) + array->offset() - (mFirstIndex >> SCALE_FACTOR);
188  mLast = mCurrent + array->length() + (mFirstIndex >> SCALE_FACTOR);
189  }
190 
191  decltype(auto) operator*() const
192  {
193  if constexpr (ChunkingPolicy::chunked) {
194  if (O2_BUILTIN_UNLIKELY(((mCurrent + (*mCurrentPos >> SCALE_FACTOR)) >= mLast))) {
195  nextChunk();
196  }
197  }
198  if constexpr (std::is_same_v<bool, std::decay_t<T>>) {
199  // FIXME: notice that due to the type punning we cannot simply convert the
200  // masked char to a bool, because it's undefined behavior.
201  // FIXME: check if shifting the masked bit to the first position is better than != 0
202  return (*((char*)mCurrent + (*mCurrentPos >> SCALE_FACTOR)) & (1 << (*mCurrentPos & 0x7))) != 0;
203  } else {
204  return *(mCurrent + (*mCurrentPos >> SCALE_FACTOR));
205  }
206  }
207 
208  // Move to the chunk which containts element pos
209  ColumnIterator<T>& moveToPos()
210  {
211  // If we get outside range of the current chunk, go to the next.
212  if constexpr (ChunkingPolicy::chunked) {
213  while (O2_BUILTIN_UNLIKELY((mCurrent + (*mCurrentPos >> SCALE_FACTOR)) >= mLast)) {
214  nextChunk();
215  }
216  }
217  return *this;
218  }
219 
220  // Move to the chunk which containts element pos
221  ColumnIterator<T>& checkNextChunk()
222  {
223  if constexpr (ChunkingPolicy::chunked) {
224  if (O2_BUILTIN_LIKELY((mCurrent + (*mCurrentPos >> SCALE_FACTOR)) <= mLast)) {
225  return *this;
226  }
227  nextChunk();
228  }
229  return *this;
230  }
231 
232  mutable T const* mCurrent;
233  int64_t const* mCurrentPos;
234  mutable T const* mLast;
235  BackendColumnType const* mColumn;
236  mutable int mFirstIndex;
237  mutable int mCurrentChunk;
238 };
