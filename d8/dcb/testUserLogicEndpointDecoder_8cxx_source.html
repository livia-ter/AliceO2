52 SampaHeader createHeader(const std::vector<SampaCluster>& clusters)
53 {
54  uint16_t n10{0};
55  for (auto c : clusters) {
56  n10 += c.nof10BitWords();
57  }
58  SampaHeader sh;
59  sh.nof10BitWords(n10);
60  sh.packetType(SampaPacketType::Data);
61  sh.hammingCode(computeHammingCode(sh.uint64()));
62  sh.headerParity(computeHeaderParity(sh.uint64()));

91 std::vector<uint64_t> b10to64(std::vector<uint10_t> b10, uint16_t prefix14)
92 {
93  uint64_t prefix = prefix14;
94  prefix <<= 50;
95  std::vector<uint64_t> b64;
96 
97  while (b10.size() % 5) {
98  b10.emplace_back(0);
99  }
100  for (auto i = 0; i < b10.size(); i += 5) {
101  uint64_t v = build64(b10[i + 4], b10[i + 3], b10[i + 2], b10[i + 1], b10[i + 0]);
102  b64.emplace_back(v | prefix);
103  }
104  return b64;
105 }

163 constexpr uint8_t CHIP = 5;
164 constexpr uint8_t CHANNEL = 31;
165 constexpr uint16_t PREFIX = 24;
166 // exact value not relevant as long as it is non-zero,
167 // and is not setting the error bits that are used (for the moment only bit 52)
168 // Idea being to populate bits 50-63 with some data to ensure
169 // the decoder is only using the lower 50 bits to get the sync and
170 // header values, for instance.
171 
172 std::vector<uint10_t> createBuffer10(const std::vector<SampaCluster>& clustersFirstChannel,
173  const std::vector<SampaCluster>& clustersSecondChannel = {})
174 {
175  bool sync{true};
176  auto b10 = createBuffer10(clustersFirstChannel, CHIP, CHANNEL, sync);
177  if (clustersSecondChannel.size()) {
178  auto chip2 = CHIP;
179  auto ch2 = CHANNEL / 2;
180  auto b10_2 = createBuffer10(clustersSecondChannel, chip2, ch2, !sync);
181  b10.insert(b10.end(), b10_2.begin(), b10_2.end());
182  }
183  return b10;
184 }

186 template <typename CHARGESUM>
187 std::string testPayloadDecode(const std::vector<SampaCluster>& clustersFirstChannel,
188  const std::vector<SampaCluster>& clustersSecondChannel = {},
189  std::optional<size_t> insertSync = std::nullopt)
190 {
191  auto b10 = createBuffer10(clustersFirstChannel, clustersSecondChannel);
192  auto b64 = b10to64(b10, PREFIX);
193  if (insertSync.has_value() && insertSync.value() < b64.size()) {
194  b64.insert(b64.begin() + insertSync.value(), (static_cast<uint64_t>(PREFIX) << 50) | sampaSyncWord);
195  }
196  std::string results;
197 
198  uint16_t dummySolarId{0};
199  uint8_t dummyGroup{0};
200  uint8_t index = (CHIP - (CHANNEL > 32)) / 2;
201  DsElecId dsId{dummySolarId, dummyGroup, index};
202  typename UserLogicEndpointDecoder<CHARGESUM>::ElinkDecoder dec(dsId, handlePacket(results));
203  decodeBuffer<CHARGESUM>(dec, b64);
204  return results;
205 }
