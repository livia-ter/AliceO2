166 GPUdi() int TPCFastSpaceChargeCorrection::getCorrection(int slice, int row, float u, float v, float& dx, float& du, float& dv) const
167 {
168  const SplineType& spline = getSpline(slice, row);
169  const float* splineData = getSplineData(slice, row);
170  float su = 0, sv = 0;
171  mGeo.convUVtoScaledUV(slice, row, u, v, su, sv);
172  su *= spline.getGridU1().getUmax();
173  sv *= spline.getGridU2().getUmax();
174  float dxuv[3];
175  spline.interpolateU(splineData, su, sv, dxuv);
176  dx = dxuv[0];
177  du = dxuv[1];
178  dv = dxuv[2];
179  return 0;
180 }
