71 template <typename RDH, typename FORMAT, typename CHARGESUM>
72 class PageDecoderImpl
73 {
74  public:
75  PageDecoderImpl(SampaChannelHandler sampaChannelHandler, FeeLink2SolarMapper fee2solar) : mSampaChannelHandler{sampaChannelHandler},
76  mFee2SolarMapper(fee2solar)
77  {
78  }
79 
80  void operator()(Page page)
81  {
82  auto rdh = createRDH<RDH>(page);
83  FeeLinkId feeLinkId(rdhFeeId(rdh) & CRUID_MASK, rdhLinkId(rdh));
84 
85  auto p = mPayloadDecoders.find(feeLinkId);
86 
87  if (p == mPayloadDecoders.end()) {
88  mPayloadDecoders.emplace(feeLinkId, PayloadDecoderImpl<FORMAT, CHARGESUM>()(feeLinkId, mSampaChannelHandler, mFee2SolarMapper));
89  p = mPayloadDecoders.find(feeLinkId);
90  }
91 
92  uint32_t orbit = rdhOrbit(rdh);
93  p->second.process(orbit, page.subspan(sizeof(rdh), rdhPayloadSize(rdh)));
94  }
95 
96  private:
97  SampaChannelHandler mSampaChannelHandler;
98  FeeLink2SolarMapper mFee2SolarMapper;
99  std::map<FeeLinkId, typename PayloadDecoderImpl<FORMAT, CHARGESUM>::type> mPayloadDecoders;
100 };

102 template <typename RDH>
103 class PageParser
104 {
105  public:
106  void operator()(RawBuffer buffer, PageDecoder pageDecoder)
107  {
108  size_t pos{0};
109  while (pos < buffer.size_bytes() - sizeof(RDH)) {
110  auto rdh = createRDH<RDH>(buffer.subspan(pos, sizeof(RDH)));
111  auto payloadSize = rdhPayloadSize(rdh);
112  pageDecoder(buffer.subspan(pos, sizeof(RDH) + payloadSize));
113  pos += rdhOffsetToNext(rdh);
114  }
115  }
116 };

128 PageDecoder createPageDecoder(RawBuffer rdhBuffer, SampaChannelHandler channelHandler, FeeLink2SolarMapper fee2solar)
129 {
130  auto rdh = createRDH<V4>(rdhBuffer);
131  if (isValid(rdh)) {
132  if (rdhLinkId(rdh) == 15) {
133  if (rdhFeeId(rdh) & impl::CHARGESUM_MASK) {
134  return impl::PageDecoderImpl<V4, UserLogicFormat, ChargeSumMode>(channelHandler, fee2solar);
135  } else {
136  return impl::PageDecoderImpl<V4, UserLogicFormat, SampleMode>(channelHandler, fee2solar);
137  }
138  } else {
139  if (rdhFeeId(rdh) & impl::CHARGESUM_MASK) {
140  return impl::PageDecoderImpl<V4, BareFormat, ChargeSumMode>(channelHandler, fee2solar);
141  } else {
142  return impl::PageDecoderImpl<V4, BareFormat, SampleMode>(channelHandler, fee2solar);
143  }
144  }
145  }
146  throw std::invalid_argument("do not know how to create a page decoder for this RDH type\n");
147 }
